public class Student
{
    public string Name { get; set; }
    public int Age { get; set; }
    public string Group { get; set; }
    
    public Student(string name, int age, string group)
    {
        Name = name;
        Age = age;
        Group = group;
    }
    
    public void Study()
    {
        Console.WriteLine($"Студент по имени {Name}, которому {Age} лет, учится в группе {Group}");
    }
}

public class Bachelor : Student
{
    public Bachelor(string name, int age, string group) : base(name, age, group)
    {
    }
    
    public void TakeExams()
    {
        Console.WriteLine($"Бакалавр {Name} сдает экзамены в группе {Group}");
        Console.WriteLine($"Текущие экзамены: математика, физика, программирование");
    }
}
public class Master : Student
{
    public string ThesisTopic { get; set; }
    
    public Master(string name, int age, string group, string thesisTopic) : base(name, age, group)
    {
        ThesisTopic = thesisTopic;
    }
    
    public void DefendThesis()
    {
        Console.WriteLine($"Магистр {Name} защищает дипломную работу на тему: \"{ThesisTopic}\"");
        Console.WriteLine("Защита проходит перед комиссией из 5 профессоров");
    }
}
